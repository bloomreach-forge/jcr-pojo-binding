93      @Test
94      public void testQueryByXPath() throws Exception {
95          assertEquals(liveNews1, handleNode.queryObjectByXPath("nodes[@primaryType='myhippoproject:news'][1]"));
96          assertEquals(liveNews1, handleNode.queryObjectByXPath("nodes[@mixinTypes='myhippoproject:taggable'][1]"));
97          assertEquals(liveNews1, handleNode.queryObjectByXPath("nodes[@mixinTypes='myhippoproject:classifiable'][1]"));
98          assertEquals(liveNews1, handleNode.queryObjectByXPath("nodes[properties[@itemName='hippostd:state']/value='published']"));
99  
100         assertEquals(previewNews1, handleNode.queryObjectByXPath("nodes[@primaryType='myhippoproject:news'][2]"));
101         assertEquals(previewNews1, handleNode.queryObjectByXPath("nodes[@mixinTypes='myhippoproject:taggable'][2]"));
102         assertEquals(previewNews1, handleNode.queryObjectByXPath("nodes[@mixinTypes='myhippoproject:classifiable'][2]"));
103         assertEquals(previewNews1, handleNode.queryObjectByXPath("nodes[properties[@itemName='hippostd:state']/value='unpublished']"));
104 
105         ContentProperty contentProp = (ContentProperty) liveNews1.queryObjectByXPath("properties[@itemName='myhippoproject:title']");
106         assertEquals("News 1", contentProp.getValue());
107 
108         contentProp = (ContentProperty) liveNews1.queryObjectByXPath("properties[@itemName='myhippoproject:summary']");
109         assertEquals("Summary of News 1", contentProp.getValue());
110 
111         contentProp = (ContentProperty) liveNews1.queryObjectByXPath("properties[@itemName='myhippoproject:date']");
112         assertEquals("2015-12-01T12:34:56-05:00", contentProp.getValue());
113 
114         ContentNode contentNode = (ContentNode) liveNews1.queryObjectByXPath("nodes[1]");
115         assertEquals("myhippoproject:body", contentNode.getName());
116         assertEquals(liveNews1.queryObjectByXPath("nodes[1]"), liveNews1.queryObjectByXPath("nodes[@primaryType='hippostd:html']"));
117         assertEquals(liveNews1.queryObjectByXPath("nodes[1]"), liveNews1.queryObjectByXPath("nodes[@itemName='myhippoproject:body']"));
118 
119         contentProp = (ContentProperty) liveNews1.queryObjectByXPath("nodes[1]/properties[@itemName='hippostd:content']");
120         assertEquals("<p>Hello, World!</p>", contentProp.getValue());
121 
122         String content = (String) liveNews1.queryObjectByXPath("nodes[1]/properties[@itemName='hippostd:content']/value");
123         assertEquals("<p>Hello, World!</p>", content);
124 
125         contentNode = (ContentNode) liveNews1.queryObjectByXPath("nodes[2]");
126         assertEquals("myhippoproject:related", contentNode.getName());
127         assertEquals(liveNews1.queryObjectByXPath("nodes[2]"), liveNews1.queryObjectByXPath("nodes[@primaryType='hippo:mirror']"));
128         assertEquals(liveNews1.queryObjectByXPath("nodes[2]"), liveNews1.queryObjectByXPath("nodes[@itemName='myhippoproject:related']"));
129 
130         contentNode = (ContentNode) liveNews1.queryObjectByXPath("nodes[3]");
131         assertEquals("myhippoproject:related", contentNode.getName());
132         assertEquals(liveNews1.queryObjectByXPath("nodes[3]"), liveNews1.queryObjectByXPath("nodes[@primaryType='hippo:mirror'][2]"));
133         assertEquals(liveNews1.queryObjectByXPath("nodes[3]"), liveNews1.queryObjectByXPath("nodes[@itemName='myhippoproject:related'][2]"));
134     }

146         log.debug("documentContent: {}", documentContent);

152     @Test
153     public void testDocumentContentSerializationInJAXB() throws Exception {
154         JAXBContext jaxbContext = JAXBContext.newInstance(ContentNode.class);
155         Marshaller jaxbMarshaller = jaxbContext.createMarshaller();
156         jaxbMarshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
157         StringWriter sw = new StringWriter();
158         jaxbMarshaller.marshal(liveNews1, sw);
159         String xmlString = sw.toString();
160         log.debug("xmlString from liveNews1: {}", xmlString);
161 
162         sw = new StringWriter();
163         jaxbMarshaller.marshal(previewNews1, sw);
164         xmlString = sw.toString();
165         log.debug("xmlString from previewNews1: {}", xmlString);
166 
167         Unmarshaller jaxbUnmarshaller = jaxbContext.createUnmarshaller();
168         ContentNode documentContent = (ContentNode) jaxbUnmarshaller.unmarshal(new StringReader(xmlString));
169         log.debug("documentContent: {}", documentContent);
170 
171         sw = new StringWriter();
172         jaxbMarshaller.marshal(documentContent, sw);
173         xmlString = sw.toString();
174         log.debug("xmlString from documentContent: {}", xmlString);
175     }

177     @Test
178     public void testSameDocumentContentSerializationInEitherJsonOrJAXB() throws Exception {
179         ObjectMapper mapper = new ObjectMapper();
180         String jsonString = mapper.writerWithDefaultPrettyPrinter().writeValueAsString(liveNews1);
181         log.debug("jsonString: {}", jsonString);
182         ContentNode documentContentFromJSON = mapper.readValue(jsonString, ContentNode.class);
183 
184         JAXBContext jaxbContext = JAXBContext.newInstance(ContentNode.class);
185         Marshaller jaxbMarshaller = jaxbContext.createMarshaller();
186         jaxbMarshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
187         StringWriter sw = new StringWriter();
188         jaxbMarshaller.marshal(liveNews1, sw);
189         String xmlString = sw.toString();
190         log.debug("xmlString: {}", xmlString);
191         Unmarshaller jaxbUnmarshaller = jaxbContext.createUnmarshaller();
192         ContentNode documentContentFromJaxb = (ContentNode) jaxbUnmarshaller.unmarshal(new StringReader(xmlString));
193 
194         assertEquals(documentContentFromJSON, documentContentFromJaxb);
195     }

197     @Test
198     public void testContentNodeIndex() throws Exception {
199         ContentNode root = new ContentNode("", "repo:root");
200         assertNull(root.getParent());
201         assertFalse(root.hasAnyNode());
202         assertEquals(1, root.getIndex());
203 
204         ContentNode folder1 = new ContentNode("folder1", "nt:folder");
205         ContentNode folder2 = new ContentNode("folder2", "nt:folder");
206         ContentNode folder3_1 = new ContentNode("folder3", "nt:folder");
207         ContentNode folder3_2 = new ContentNode("folder3", "nt:folder");
208         ContentNode folder3_3 = new ContentNode("folder3", "nt:folder");
209         ContentNode folder6 = new ContentNode("folder6", "nt:folder");
210         ContentNode folder7_1 = new ContentNode("folder7", "nt:folder");
211         ContentNode folder7_2 = new ContentNode("folder7", "nt:folder");
212         ContentNode folder9 = new ContentNode("folder9", "nt:folder");
213 
214         root.addNode(folder1);
215         root.addNode(folder2);
216         root.addNode(folder3_1);
217         root.addNode(folder3_2);
218         root.addNode(folder3_3);
219         root.addNode(folder6);
220         root.addNode(folder7_1);
221         root.addNode(folder7_2);
222         root.addNode(folder9);
223 
224         assertEquals(1, folder1.getIndex());
225         assertEquals(1, folder2.getIndex());
226         assertEquals(1, folder3_1.getIndex());
227         assertEquals(2, folder3_2.getIndex());
228         assertEquals(3, folder3_3.getIndex());
229         assertEquals(1, folder6.getIndex());
230         assertEquals(1, folder7_1.getIndex());
231         assertEquals(2, folder7_2.getIndex());
