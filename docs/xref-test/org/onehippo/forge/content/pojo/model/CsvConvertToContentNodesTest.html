16  package org.onehippo.forge.content.pojo.model;
17  
18  import java.io.InputStream;
19  import java.io.InputStreamReader;
20  import java.io.StringWriter;
21  import java.net.URL;
22  import java.util.Iterator;
23  import java.util.UUID;
24  
25  import org.apache.commons.csv.CSVFormat;
26  import org.apache.commons.csv.CSVParser;
27  import org.apache.commons.csv.CSVRecord;
28  import org.apache.tika.io.IOUtils;
29  import org.hippoecm.repository.HippoStdNodeType;
30  import org.hippoecm.repository.api.StringCodec;
31  import org.hippoecm.repository.api.StringCodecFactory;
32  import org.hippoecm.repository.translation.HippoTranslationNodeType;
33  import org.junit.Test;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import com.fasterxml.jackson.databind.ObjectMapper;
38  
39  public class CsvConvertToContentNodesTest {
40  
41      private static Logger log = LoggerFactory.getLogger(CsvConvertToContentNodesTest.class);
42  
43      private URL NEWS_CSV_URL = CsvConvertToContentNodesTest.class.getResource("news.csv");
44  
45      @Test
46      public void testReadCsvAndConvertToContentNodes() throws Exception {
47          InputStream input = null;
48          InputStreamReader reader = null;
49  
50          try {
51              // 1. Open a reader from a CSV file.
52              input = NEWS_CSV_URL.openStream();
53              reader = new InputStreamReader(input, "UTF-8");
54  
55              // 2. Create CSV parser to parse the CSV data with column headers.
56              CSVParser parser = CSVFormat.DEFAULT.withHeader("Title", "Introduction", "Date", "Content")
57                      .withSkipHeaderRecord().parse(reader);
58              CSVRecord record;
59  
60              // 3. StringCodec to generate a JCR node name from the title column,
61              //    and ObjectMapper instance to log a ContentNode to JSON.
62              final StringCodec codec = new StringCodecFactory.UriEncoding();
63              final ObjectMapper objectMapper = new ObjectMapper();
64  
65              String name;
66              String title;
67              String introduction;
68              String date;
69              String content;
70  
71              String translationId;
72              String translationLocale = "en";
73  
74              String targetDocumentLocation;
75  
76              // 4. Iterate each data record and create a ContentNode for a news article with setting properties and child nodes.
77              for (Iterator<CSVRecord> it = parser.iterator(); it.hasNext(); ) {
78                  record = it.next();
79  
80                  // 4.1. Read each column from a CSV record.
81                  title = record.get("Title");
82                  name = codec.encode(title);
83                  introduction = record.get("Introduction");
84                  date = record.get("Date");
85                  content = record.get("Content");
86  
87                  // 4.2. Create a ContentNode for a news article and set primitive property values.
88                  ContentNode newsNode = new ContentNode(name, "ns1:newsdocument");
89                  newsNode.setProperty("ns1:title", title);
90                  newsNode.setProperty("ns1:introduction", introduction);
91                  newsNode.setProperty("ns1:date", ContentPropertyType.DATE, date);
92  
93                  // 4.3. Create/add a child hippostd:html content node and set the content.
94                  ContentNode htmlNode = new ContentNode("ns1:content", HippoStdNodeType.NT_HTML);
95                  newsNode.addNode(htmlNode);
96                  htmlNode.setProperty(HippoStdNodeType.HIPPOSTD_CONTENT, content);
97  
98                  // 4.4. In Hippo CMS, the internal translation UUID and locale string are important in most cases.
99                  //      So, let's generate a translation UUID and use 'en' for simplicity for now.
100                 translationId = UUID.randomUUID().toString();
101                 newsNode.setProperty(HippoTranslationNodeType.ID, translationId);
102                 newsNode.setProperty(HippoTranslationNodeType.LOCALE, translationLocale);
103 
104                 // 4.5. (Optional) Set kind of meta property for localized document name which is displayed in folder view later.
105                 //      This meta property is not used by Hippo CMS, but can be read/used by a higher level content importing application
106                 //      to set a localized (translated) name of the document (e.g, using Hippo TranslationWorkflow).
107                 newsNode.setProperty("jcr:localizedName", title);
108 
109                 // 4.6. (Optional) Determine the target document location where this content should be generated and
110                 //      store it in a meta property, jcr:path.
111                 //      This meta property cannot be used in JCR repository in importing process, but can be read/used by a higher level
112                 //      content importing application to create a document using Hippo DocumentWorkflow for instance.
113                 targetDocumentLocation = "/content/documents/ns1/news/" + name;
114                 newsNode.setProperty("jcr:path", targetDocumentLocation);
115 
116                 // 4.7. (Optional) Log the JSON-ized string of the news ContentNode instance.
117                 StringWriter stringWriter = new StringWriter(256);
118                 objectMapper.writerWithDefaultPrettyPrinter().writeValue(stringWriter, newsNode);
119                 log.debug("newsNode: \n{}\n", stringWriter.toString());
120             }
121         } finally {
122             IOUtils.closeQuietly(reader);
123             IOUtils.closeQuietly(input);
124         }
125     }
