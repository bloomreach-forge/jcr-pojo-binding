111     protected MockNode createHippoGalleryFolderNode(MockNode baseNode, String name, String translationName,
112             String translationLanguage) throws Exception {
113         MockNode galleryFolderNode = baseNode.addNode(name, "hippogallery:stdImageGallery");
114         galleryFolderNode.addMixin("mix:referenceable");
115         galleryFolderNode.setProperty("hippostd:foldertype", new String[] { "new-image-folder" });
116         galleryFolderNode.setProperty("hippostd:gallerytype", new String[] { "hippogallery:imageset" });
117 
118         if (translationName != null) {
119             MockNode translationNode = galleryFolderNode.addNode(HippoNodeType.HIPPO_TRANSLATION,
120                     HippoNodeType.NT_TRANSLATION);
121             translationNode.setProperty(HippoNodeType.HIPPO_LANGUAGE, translationLanguage);
122             translationNode.setProperty(HippoNodeType.HIPPO_MESSAGE, translationName);
123         }
124 
125         return galleryFolderNode;
126     }

152     protected MockNode createHippoFolderNode(MockNode baseNode, String name, String translationName,
153             String translationLanguage) throws Exception {
154         MockNode folderNode = baseNode.addNode(name, HippoStdNodeType.NT_FOLDER);
155         folderNode.addMixin("hippo:translated");
156         folderNode.addMixin("mix:referenceable");
157         folderNode.setProperty("hippostd:foldertype", new String[] { "new-translated-folder", "new-document" });
158 
159         if (translationName != null) {
160             MockNode translationNode = folderNode.addNode(HippoNodeType.HIPPO_TRANSLATION,
161                     HippoNodeType.NT_TRANSLATION);
162             translationNode.setProperty(HippoNodeType.HIPPO_LANGUAGE, translationLanguage);
163             translationNode.setProperty(HippoNodeType.HIPPO_MESSAGE, translationName);
164         }
165 
166         return folderNode;
167     }

172         handleNode.addMixin("mix:referenceable");
173         handleNode.addMixin(HippoNodeType.NT_TRANSLATED);
174

176             MockNode translationNode = handleNode.addNode(HippoNodeType.HIPPO_TRANSLATION,
177                     HippoNodeType.NT_TRANSLATION);

179             translationNode.setProperty(HippoNodeType.HIPPO_MESSAGE, translationName);
180         }
181 
182         return handleNode;
183     }

185     protected MockNode createHippoDocumentVariantNode(MockNode handleNode, String primaryType, String state)
186             throws Exception {
187         MockNode variantNode = handleNode.addNode(handleNode.getName(), primaryType);
188 
189         for (String mixin : DOC_VARIANT_NODE_MIXIN_TYPES) {
190             variantNode.addMixin(mixin);
191         }
192 
193         if (HippoStdNodeType.PUBLISHED.equals(state)) {
194             variantNode.setProperty(HippoNodeType.HIPPO_AVAILABILITY, new String[] { "live", "preview" });
195             variantNode.setProperty(HippoStdNodeType.HIPPOSTD_STATE, HippoStdNodeType.PUBLISHED);
196             variantNode.setProperty(HippoStdNodeType.HIPPOSTD_STATESUMMARY, "live");
197         } else {
198             variantNode.setProperty(HippoNodeType.HIPPO_AVAILABILITY, new String[] { "preview" });
199             variantNode.setProperty(HippoStdNodeType.HIPPOSTD_STATE, HippoStdNodeType.UNPUBLISHED);
200             variantNode.setProperty(HippoStdNodeType.HIPPOSTD_STATESUMMARY, "changed");
201         }
202 
203         return variantNode;
204     }
