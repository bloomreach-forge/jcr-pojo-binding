17  import java.util.HashSet;
18  import java.util.Set;
19  
20  import javax.jcr.Node;
21  import javax.jcr.RepositoryException;
22  
23  import org.hippoecm.repository.HippoStdNodeType;
24  import org.hippoecm.repository.api.HippoNodeType;
25  import org.onehippo.forge.content.pojo.mapper.ContentNodeMappingException;
26  import org.onehippo.forge.content.pojo.mapper.ContentNodeMappingItemFilter;
27  
28  /**
29   * A {@link ContentNodeMappingItemFilter} implementation for Hippo JCR {@link javax.jcr.Item},
30   * extending {@link DefaultHippoJcrItemMappingFilter},
31   * to select given document variant types ({@link HippoStdNodeType#HIPPOSTD_STATE} property value) by default.
32   */
33  public class DocumentVariantNodeMappingFilter extends DefaultHippoJcrItemMappingFilter {
34  
35      private Set<String> expectedStateSet;
36  
37      public DocumentVariantNodeMappingFilter(String... expectedStates) {
38          super();
39  
40          expectedStateSet = new HashSet<>();
41  
42          if (expectedStates != null && expectedStates.length != 0) {
43              for (String expectedState : expectedStates) {
44                  expectedStateSet.add(expectedState);
45              }
46          }
47      }
48  
49      @Override
50      protected boolean acceptNode(Node node) throws ContentNodeMappingException {
51          if (!super.acceptNode(node)) {
52              return false;
53          }
54  
55          try {
56              if (isDocumentVariantNode(node)) {
57                  if (node.hasProperty(HippoStdNodeType.HIPPOSTD_STATE)
58                          && expectedStateSet.contains(node.getProperty(HippoStdNodeType.HIPPOSTD_STATE).getString())) {
59                      return true;
60                  } else {
61                      return false;
62                  }
63              } else {
64                  return true;
65              }
66          } catch (RepositoryException e) {
67              throw new ContentNodeMappingException(e.toString(), e);
68          }
69      }
70  
71      protected boolean isDocumentVariantNode(final Node jcrNode) throws RepositoryException {
72          if (jcrNode.isNodeType(HippoNodeType.NT_DOCUMENT)
73                  && jcrNode.isNodeType(HippoStdNodeType.NT_PUBLISHABLE)
74                  && jcrNode.getParent().isNodeType(HippoNodeType.NT_HANDLE)) {
75              return true;
76          }
77  
78          return false;
79      }
80  
81  }
