16  package org.onehippo.forge.content.pojo.mapper;
17  
18  import java.io.Serializable;
19  
20  import org.onehippo.forge.content.pojo.common.ContentValueConverter;
21  import org.onehippo.forge.content.pojo.model.ContentNode;
22  
23  /**
24   * Content Node Mapper interface, mapping from a physical <code>dataNode</code> to a {@link ContentNode} instance.
25   *
26   * @param <D> physical data node (e.g, {@link javax.jcr.Node}) from which a {@link ContentNode} should be converted to.
27   * @param <I> physical data item (e.g, {@link javax.jcr.Item}) to be used in filtering in conversion.
28   * @param <V> value converter from the physical data value (e.g, {@link javax.jcr.Value}) to be used in converting a physical value.
29   */
30  public interface ContentNodeMapper<D, I, V> extends Serializable {
31  
32      /**
33       * Maps {@code dataNode} to a {@link ContentNode}.
34       * @param dataNode physical data node
35       * @return a converted {@link ContentNode}
36       * @throws ContentNodeMappingException if any content node mapping exception occurs
37       */
38      public ContentNode map(D dataNode) throws ContentNodeMappingException;
39  
40      /**
41       * Maps {@code dataNode} to a {@link ContentNode} using {@code itemFilter}.
42       * @param dataNode physical data node
43       * @param itemFilter filter for a physical data item
44       * @return a converted {@link ContentNode}
45       * @throws ContentNodeMappingException if any content node mapping exception occurs
46       */
47      public ContentNode map(D dataNode, ContentNodeMappingItemFilter<I> itemFilter) throws ContentNodeMappingException;
48  
49      /**
50       * Maps {@code dataNode} to a {@link ContentNode} using {@code itemFilter} and {@code valueConverter}.
51       * @param dataNode physical data node
52       * @param itemFilter filter for a physical data item
53       * @param valueConverter converter used in converting a physical data value
54       * @return a converted {@link ContentNode}
55       * @throws ContentNodeMappingException if any content node mapping exception occurs
56       */
57      public ContentNode map(D dataNode, ContentNodeMappingItemFilter<I> itemFilter, ContentValueConverter<V> valueConverter) throws ContentNodeMappingException;
58  
59  }
