49  /**
50   * Default {@link ContentValueConverter} implementation for JCR.
51   * <P>
52   * Especially for binary data, this converts JCR {@link javax.jcr.Binary} value to
53   * a <code>data:</code> URL if the data size is less than {@link #getDataUrlSizeThreashold()}.
54   * Otherwise, this stores the binary data in an external {@link FileObject} in a randomly generated
55   * file under the {@link #getBinaryValueFileFolder()} with the file name prefix, {@link #getBinaryFileNamePrefix()},
56   * and keeps the file URL string instead of the whole data.
57   * </P>
58   */

121     @Override
122     public String toString(Value value) throws ContentNodeException {
123         try {
124             String stringifiedValue = null;
125 
126             switch (value.getType()) {
127             case PropertyType.STRING: {
128                 stringifiedValue = value.getString();
129                 break;
130             }
131             case PropertyType.LONG: {
132                 stringifiedValue = Long.toString(value.getLong());
133                 break;
134             }
135             case PropertyType.DOUBLE: {
136                 stringifiedValue = Double.toString(value.getDouble());
137                 break;
138             }
139             case PropertyType.DATE: {
140                 stringifiedValue = ISO8601.format(value.getDate());
141                 break;
142             }
143             case PropertyType.BOOLEAN: {
144                 stringifiedValue = Boolean.toString(value.getBoolean());
145                 break;
146             }
147             case PropertyType.DECIMAL: {
148                 stringifiedValue = value.getDecimal().toString();
149                 break;
150             }
151             case PropertyType.NAME:
152             case PropertyType.URI: {
153                 stringifiedValue = value.getString();
154                 break;
155             }
156             }
157 
158             return stringifiedValue;
159         } catch (RepositoryException e) {
160             throw new ContentNodeException(e.toString(), e);
161         }
162     }

164     @Override
165     public BinaryValue toBinaryValue(Value value, String mimeType) throws ContentNodeException {
166         try {
167             Binary binary = value.getBinary();
168             long size = binary.getSize();
169 
170             if (size < getDataUrlSizeThreashold()) {
171                 InputStream input = null;
172                 ByteArrayOutputStream output = null;
173 
174                 try {
175                     input = binary.getStream();
176                     output = new ByteArrayOutputStream((int) size);
177                     IOUtils.copy(input, output);
178 
179                     input.close();
180                     input = null;
181                     binary.dispose();
182                     binary = null;
183 
184                     return new BinaryValue(output.toByteArray());
185                 } finally {
186                     IOUtils.closeQuietly(output);
187                     IOUtils.closeQuietly(input);
188 
189                     if (binary != null) {
190                         binary.dispose();
191                     }
192                 }
193             } else {
194                 FileObject binaryFile = createRandomBinaryValueFileObject(mimeType);
195                 InputStream input = null;
196                 OutputStream output = null;
197 
198                 try {
199                     input = binary.getStream();
200                     output = binaryFile.getContent().getOutputStream();
201                     IOUtils.copy(input, output);
202 
203                     output.close();
204                     output = null;
205                     input.close();
206                     input = null;
207                     binary.dispose();
208                     binary = null;
209 
210                     return new BinaryValue(binaryFile);
211                 } finally {
212                     IOUtils.closeQuietly(output);
213                     IOUtils.closeQuietly(input);
214 
215                     if (binary != null) {
216                         binary.dispose();
217                     }
218                 }
219             }
220         } catch (Exception e) {
221             throw new ContentNodeException(e.toString(), e);
222         }
223     }

225     @Override
226     public Value toJcrValue(String typeName, String stringValue) throws ContentNodeException {
227         if (stringValue == null) {
228             throw new IllegalArgumentException("Cannot convert null to a JCR value.");
229         }
230 
231         ContentPropertyType type = ContentPropertyType.valueOf(typeName);
232         Value jcrValue = null;
233 
234         try {
235             final ValueFactory valueFactory = getSession().getValueFactory();
236 
237             switch (type) {
238             case STRING:
239                 jcrValue = valueFactory.createValue(stringValue);
240                 break;
241             case DATE:
242                 jcrValue = valueFactory.createValue(ISO8601.parse(stringValue));
243                 break;
244             case BOOLEAN:
245                 jcrValue = valueFactory.createValue(BooleanUtils.toBoolean(stringValue));
246                 break;
247             case LONG:
248                 jcrValue = valueFactory.createValue(NumberUtils.toLong(stringValue));
249                 break;
250             case DOUBLE:
251                 jcrValue = valueFactory.createValue(NumberUtils.toDouble(stringValue));
252                 break;
253             case DECIMAL:
254                 jcrValue = valueFactory.createValue(new BigDecimal(stringValue));
255                 break;
256             default:
257                 break;
258             }
259         } catch (RepositoryException e) {
260             throw new ContentNodeException(e.toString(), e);
261         }
262 
263         return jcrValue;
264     }

266     @Override
267     public Value toJcrValue(BinaryValue binaryValue) throws ContentNodeException {
268         InputStream input = null;
269         Binary binary = null;
270 
271         try {
272             ValueFactory valueFactory = getSession().getValueFactory();
273             input = binaryValue.getStream();
274             binary = valueFactory.createBinary(input);
275             return valueFactory.createValue(binary);
276         } catch (IOException | RepositoryException e) {
277             IOUtils.closeQuietly(input);
278             throw new ContentNodeException(e.toString(), e);
279         }
280     }

286     protected FileObject createRandomBinaryValueFileObject(final String mimeType) throws IOException {
287         String fileNameSuffix = getDefaultBinaryFileNameSuffix();
288         String extension = findDefaultFileExtensionByMimeType(mimeType);
289 
290         if (StringUtils.isNotEmpty(extension)) {
291             fileNameSuffix = extension;
292         }
293 
294         if (getBinaryValueFileFolder() == null) {
295             File binaryFile = File.createTempFile(getBinaryFileNamePrefix(), fileNameSuffix);
296             return VFS.getManager().toFileObject(binaryFile);
297         } else {
298             getBinaryValueFileFolder().createFolder();
299             FileObject binaryFileObject = getBinaryValueFileFolder()
300                     .resolveFile(getBinaryFileNamePrefix() + System.currentTimeMillis() + fileNameSuffix);
301             return binaryFileObject;
302         }
303     }
304 
305     protected String findDefaultFileExtensionByMimeType(final String mimeTypeValue) {
306         String extension = null;
307 
308         try {
309             if (StringUtils.isNotEmpty(mimeTypeValue)) {
310                 final MimeType mimeType = getMimeTypes().forName(mimeTypeValue);
311                 extension = mimeType.getExtension();
312             }
313         } catch (MimeTypeException e) {
314             log.warn("Failed to determine a file extension by the mimeType: '{}'", mimeTypeValue, e);
315         }
316 
317         return StringUtils.trim(extension);
318     }
