289     /**
290      * Converts the internally stored stringified value at the value {@code index} to a native Java object
291      * such as {@link String}, {@link Calendar}, {@link Boolean}, {@link Long}, {@link Double}, {@link BigDecimal} and {@link BinaryValue}
292      * based on the {@link #getType()} value of this content property.
293      * @param index value index
294      * @return converted native Java object from the internal stringified value based on content property type
295      */
296     private Object getValueAsObjectAt(final int index) {
297         Object objectValue = null;
298 
299         final String stringifiedValue = values.get(index);
300 
301         switch (type) {
302         case STRING: {
303             objectValue = stringifiedValue;
304             break;
305         }
306         case DATE: {
307             objectValue = ISO8601.parse(stringifiedValue);
308             break;
309         }
310         case BOOLEAN: {
311             objectValue = Boolean.parseBoolean(stringifiedValue);
312             break;
313         }
314         case LONG: {
315             objectValue = Long.parseLong(stringifiedValue);
316             break;
317         }
318         case DOUBLE: {
319             objectValue = Double.parseDouble(stringifiedValue);
320             break;
321         }
322         case DECIMAL: {
323             objectValue = new BigDecimal(stringifiedValue);
324             break;
325         }
326         case BINARY: {
327             objectValue = createBinaryValue(stringifiedValue);
328             break;
329         }
330         default: {
331             throw new UnsupportedOperationException(
332                     "Unsupported type, " + type + ". Only primitive number/string values are currently supported.");
333         }
334         }
335 
336         return objectValue;
337     }

339     /**
340      * Creates a {@link BinaryValue} instance from the given stringified value
341      * which can be either a <code>data:</code> URL or any other external URL to be read.
342      * @param stringifiedValue stringfieid binary value, either a <code>data:</code> URL or any other external URL to be read
343      * @return a {@link BinaryValue} instance
344      */
345     private BinaryValue createBinaryValue(String stringifiedValue) {
346         BinaryValue binaryValue = null;
347 
348         if (StringUtils.startsWith(stringifiedValue, "data:")) {
349             binaryValue = BinaryValue.fromDataURI(stringifiedValue);
350         } else {
351             try {
352                 FileSystemManager fsManager = VFS.getManager();
353                 FileObject fileObject = fsManager.resolveFile(stringifiedValue);
354                 binaryValue = new BinaryValue(fileObject);
355             } catch (FileSystemException e) {
356                 throw new IllegalArgumentException("Unresolvable VFS url.");
357             }
358         }
359 
360         return binaryValue;
361     }
